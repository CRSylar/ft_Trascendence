## Comunicare con API INTRA
```
import { Controller, Get, Param } from '@nestjs/common';
import { AppService } from './app.service';
import fetch from 'node-fetch';
import { response } from 'express';

interface iToken {
  token: string,
  expire_at: Date
}

var token: iToken = {token: "", expire_at: new Date()};

function getToken(): any {
  return fetch('https://api.intra.42.fr/oauth/token', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      'grant_type': 'client_credentials',
      'client_id': '7bd40ef252061c4be4ac1d72d493117027656e2321f8a7320e23ee00485b7e3a',
      'client_secret': 'bed23846031f7ddc03fb3889eea3b370ad29a6f8a8088688ef9529654aa5cf9f'
    })
  })
    .then(response => response.json())
    .then(data => {
      token.token = data.access_token;
      token.expire_at = new Date(new Date().getTime() + parseInt(data.expires_in));
    })
}

@Controller()
export class AppController {
  constructor(private readonly appService: AppService) {}

  @Get()
  getHello(): string {
    return this.appService.getHello();
  }

  @Get("/user/:id")
  getInfoUser(@Param('id') id:string): any {
    if (new Date() > token.expire_at)
      getToken();
    return fetch(`https://api.intra.42.fr/v2/users/${id}`, {
      method: 'GET',
      headers: {
        'Authorization': `Bearer ${token.token}`
      }
    })
      .then(response => response.json());
  }

}
```
